# RPC v2 Strict Cutover Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the current JSONL RPC wire format with the v2 schema that separates push events from pull responses, supports request IDs, and surfaces malformed lines in the TUI.

**Architecture:** The agent emits push events without a `kind` field and pull responses wrapped in `kind: "response"`. The TUI parser routes messages by checking `kind` and exposes unknown/malformed lines in chat. Strict cutover: only v2 is accepted.

**Tech Stack:** Rust (ratatui, crossterm, serde), TypeScript (bun, Vercel AI SDK, OpenRouter provider)

```ts
// agent/src/rpc.ts
export type PushEvent =
  | { type: 'agent_start' }
  | { type: 'turn_start' }
  | { type: 'text_delta'; delta: string }
  | { type: 'turn_end' }
  | { type: 'agent_end' }
  | { type: 'error'; message: string };

export type PullResponse =
  | { kind: 'response'; command: 'get_state';         id?: string; success: true;  data: StateData }
  | { kind: 'response'; command: 'get_session_stats'; id?: string; success: true;  data: SessionStatsData }
  | { kind: 'response'; command: 'prompt';            id?: string; success: true }
  | { kind: 'response'; command: 'abort';             id?: string; success: true }
  | { kind: 'response'; command: 'clear';             id?: string; success: true }
  | { kind: 'response'; command: string;              id?: string; success: false; error: string };

export interface StateData {
  model_name:  string;
  model_limit: number;
  temp:        number;
  is_streaming: boolean;
}

export interface SessionStatsData {
  tokens: {
    input:       number;
    output:      number;
    cache_read:  number;
    cache_write: number;
    total:       number;
  };
  context_usage: {
    tokens:  number;
    limit:   number;
    percent: number;
  };
  cost:  number;
  turns: number;
}

export type RpcMessage = PushEvent | PullResponse;

function writeLine(obj: unknown): void {
  process.stdout.write(JSON.stringify(obj) + '\n');
}

export function emitEvent(event: PushEvent): void {
  writeLine(event);
}

export function emitResponse(response: PullResponse): void {
  writeLine(response);
}

type LineHandler = (line: string) => void | Promise<void>;

export function readStdinLines(handler: LineHandler): void {
  let buffer = '';
  process.stdin.setEncoding('utf8');
  process.stdin.on('data', (chunk: string) => {
    buffer += chunk;
    let newline: number;
    while ((newline = buffer.indexOf('\n')) !== -1) {
      const line = buffer.slice(0, newline).replace(/\r$/, '');
      buffer = buffer.slice(newline + 1);
      if (line.trim()) handler(line);
    }
  });
  process.stdin.on('end', () => process.exit(0));
}
```

```ts
// agent/src/index.ts (sketch)
import { LLMClient, getState, getSessionStats, clearHistory } from './llm.js';
import { emitEvent, emitResponse, readStdinLines } from './rpc.js';

const llm = new LLMClient();

emitEvent({ type: 'agent_start' });

readStdinLines(async (line) => {
  let payload: any;
  try {
    payload = JSON.parse(line);
  } catch {
    emitEvent({ type: 'error', message: 'Invalid JSON' });
    return;
  }

  const id = typeof payload.id === 'string' ? payload.id : undefined;

  switch (payload.type) {
    case 'get_state':
      emitResponse({ kind: 'response', command: 'get_state', success: true, id, data: getState() });
      break;
    case 'get_session_stats':
      emitResponse({ kind: 'response', command: 'get_session_stats', success: true, id, data: getSessionStats() });
      break;
    case 'clear':
      clearHistory();
      emitResponse({ kind: 'response', command: 'clear', success: true, id });
      emitEvent({ type: 'cleared' } as any);
      break;
    case 'prompt':
      emitResponse({ kind: 'response', command: 'prompt', success: true, id });
      await llm.streamResponse(payload.content);
      break;
    case 'abort':
      llm.abort();
      emitResponse({ kind: 'response', command: 'abort', success: true, id });
      break;
    default:
      emitResponse({ kind: 'response', command: payload.type ?? 'unknown', success: false, id, error: 'Unknown command' });
  }
});
```

```ts
// agent/src/llm.ts (sketch)
let abortController: AbortController | null = null;
let totalInput = 0;
let totalOutput = 0;
let totalTurns = 0;

export class LLMClient {
  abort() {
    abortController?.abort();
  }

  async streamResponse(userMessage: string) {
    history.push({ role: 'user', content: userMessage });
    emitEvent({ type: 'turn_start' });

    abortController = new AbortController();

    try {
      const result = await streamText({
        model: openrouter.chat(MODEL_NAME),
        temperature: MODEL_TEMP,
        messages: history,
        abortSignal: abortController.signal,
      });

      let assistantText = '';
      for await (const chunk of result.textStream) {
        assistantText += chunk;
        emitEvent({ type: 'text_delta', delta: chunk });
      }

      if (assistantText) history.push({ role: 'assistant', content: assistantText });
      trimHistory();

      const usage = await result.usage;
      const input = usage?.promptTokens ?? usage?.inputTokens ?? 0;
      const output = usage?.completionTokens ?? usage?.outputTokens ?? 0;
      totalInput += input;
      totalOutput += output;
      totalTurns += 1;

    } catch (err) {
      history.pop();
      emitEvent({ type: 'error', message: `LLM error: ${String(err)}` });
      emitEvent({ type: 'agent_end' });
      return;
    }

    emitEvent({ type: 'turn_end' });
    emitEvent({ type: 'agent_end' });
  }
}

export function getSessionStats() {
  const total = totalInput + totalOutput;
  const percent = MODEL_LIMIT > 0 ? (total / MODEL_LIMIT) * 100 : 0;
  return {
    tokens: { input: totalInput, output: totalOutput, cache_read: 0, cache_write: 0, total },
    context_usage: { tokens: total, limit: MODEL_LIMIT, percent },
    cost: 0,
    turns: totalTurns,
  };
}

export function getState() {
  return {
    model_name: MODEL_NAME.split('/')[1]?.split(':')[0] || MODEL_NAME,
    model_limit: MODEL_LIMIT,
    temp: MODEL_TEMP,
    is_streaming: false,
  };
}
```

- [ ] **Step 2: Implement v2 message types and parser**

```rust
#[derive(Debug, Clone, Deserialize)]
#[serde(tag = "type")]
pub enum PushEvent { /* agent_start, turn_start, ... */ }

#[derive(Debug, Clone, Deserialize)]
#[serde(tag = "kind")]
pub enum PullResponse { /* response variants */ }

#[derive(Debug, Clone)]
pub enum AgentMsg {
    Push(PushEvent),
    Pull(PullResponse),
    Unknown { raw: String },
}

pub fn parse_line(line: &str) -> AgentMsg {
    if let Ok(pull) = serde_json::from_str::<PullResponse>(line) {
        return AgentMsg::Pull(pull);
    }
    if let Ok(push) = serde_json::from_str::<PushEvent>(line) {
        return AgentMsg::Push(push);
    }
    AgentMsg::Unknown { raw: line.to_string() }
}
```
